/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_SPI1_Init();
  MX_USB_DEVICE_Init();
  MX_I2C1_Init();
  MX_RTC_Init();

  /* Initialize interrupts */
  MX_NVIC_Init();
  /* USER CODE BEGIN 2 */
  myLoRa = newLoRa();
  myLoRa.hSPIx = &hspi1;
  myLoRa.CS_port = GPIOB;
  myLoRa.CS_pin = GPIO_PIN_0;
  myLoRa.reset_port = GPIOB;
  myLoRa.reset_pin = GPIO_PIN_1;
  myLoRa.DIO0_port = GPIOB;
  myLoRa.DIO0_pin = GPIO_PIN_10;


  while (LoRa_init(&myLoRa) != LORA_OK) {
	  cdc_print("LoRa pwr OFF\n\r");
  }
  LoRa_stat = 1;

  LiquidCrystal(GPIOB, LCD_RS_Pin, 255, LCD_E_Pin,
  		  LCD_D4_Pin, LCD_D5_Pin, LCD_D6_Pin, LCD_D7_Pin);
  HAL_Delay(50);

  LoRa_startReceiving(&myLoRa);

  air_mode = OFF;
  char* command;
  uint32_t temp_update = 0;
  uint32_t auto_temp_timer = 0;
  temperature = Read_Temperature();
  temp_setpt = 20;
  SendStatsToTx((int) temperature);
  size_t deadzone = 2;
  cdc_print("Initial Temperature: %.2f\r\n", temperature);
  LCD_StrInit();


  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  /*--------------------RECEIVER--------------------*/
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
	  HAL_Delay(50);
	  if (HAL_GetTick() - auto_temp_timer >= 5000 && !is_stopped) {
		  auto_temp_timer = HAL_GetTick();
		  temperature = Read_Temperature();
		  LCD_UpdateTemp((int) temperature);
		  SendStatsToTx((int) temperature);
		  cdc_print("Obtained temp reading: %.2f\n\r", temperature);
		  EnterStopMode();
	  }
	  if (temp_flag && HAL_GetTick() - temp_update >= 2000) {
		  cdc_print("Temp flag is up, attempting to adjust...\n\r");
		  temp_change(air_mode);
	  }

	  /*Update temperature if heat or cool is on dynamically*/
	  if ((temp_flag == 0) &&
			  (temperature < temp_setpt - deadzone && air_mode == HEAT)) {
		    cdc_print("Hit deadzone, adjusting on heat\n\r");
		    temp_flag = 1;
		    temp_update = HAL_GetTick();
	  }
	  if ((temp_flag == 0) &&
				(temperature > temp_setpt + deadzone && air_mode == AC)) {
			cdc_print("Hit deadzone, adjusting on cool\n\r");
			temp_flag = 1;
			temp_update = HAL_GetTick();
	  }

	  if (strlen((char*) RxBuffer) > 0) {
		  command = (char*) RxBuffer;
		  cdc_print("RxBuffer: %s\n\r", RxBuffer);
		  if (strcmp(command, "TEMP_UP") == 0) {
			  if (air_mode == AC || air_mode == HEAT) {
				  temp_setpt++;
				  temp_flag = 1;
				  temp_update = HAL_GetTick();
				  LCD_UpdateSetpt((int) temp_setpt);
				  cdc_print("Temperature setpoint: %d\r\n", temp_setpt);
			  }
			  else {
				  cdc_print("Neither heat or AC is turned on\r\n");
			  }
		  }
		  else if (strcmp(command, "TEMP_DOWN") == 0) {
			  if (air_mode == AC || air_mode == HEAT) {
				  temp_setpt--;
				  temp_flag = 1;
				  temp_update = HAL_GetTick();
				  LCD_UpdateSetpt((int) temp_setpt);
				  cdc_print("Temperature setpoint: %d\r\n", temp_setpt);
			  }
			  else {
				  cdc_print("Neither heat or AC is turned on\r\n");
			  }
		  }
		  else if (strcmp(command, "HEAT_ON") == 0) {
			  air_mode = HEAT;
			  LCD_UpdateAirMode(air_mode);
			  cdc_print("Turned on heat\r\n");
		  }
		  else if (strcmp(command, "AC_ON") == 0) {
			  air_mode = AC;
			  LCD_UpdateAirMode(air_mode);
			  cdc_print("Turned on AC\r\n");
		  }
		  else if (strcmp(command, "OFF_MODE") == 0) {
			  air_mode = OFF;
			  LCD_UpdateAirMode(air_mode);
			  cdc_print("Off mode toggled\r\n");
		  }
		  else {
			  cdc_print("Unknown command\r\n");
		  }
	  }

	  // Clear RxBuffer to avoid processing the same data
	  memset(RxBuffer, 0, sizeof(RxBuffer));
	  HAL_Delay(50);
  }
  /* USER CODE END 3 */
}

  * @brief System Clock Configuration
  * @retval None
  */
void SystemClock_Config(void)
{
  RCC_OscInitTypeDef RCC_OscInitStruct = {0};
  RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
  RCC_PeriphCLKInitTypeDef PeriphClkInit = {0};

  /** Initializes the RCC Oscillators according to the specified parameters
  * in the RCC_OscInitTypeDef structure.
  */
  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE|RCC_OSCILLATORTYPE_LSE;
  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
  RCC_OscInitStruct.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
  RCC_OscInitStruct.HSIState = RCC_HSI_ON;
  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL9;
  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
  {
    Error_Handler();
  }

  /** Initializes the CPU, AHB and APB buses clocks
  */
  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK
                              |RCC_CLOCKTYPE_PCLK1|RCC_CLOCKTYPE_PCLK2;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;

  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2) != HAL_OK)
  {
    Error_Handler();
  }
  PeriphClkInit.PeriphClockSelection = RCC_PERIPHCLK_RTC|RCC_PERIPHCLK_USB;
  PeriphClkInit.RTCClockSelection = RCC_RTCCLKSOURCE_LSE;
  PeriphClkInit.UsbClockSelection = RCC_USBCLKSOURCE_PLL_DIV1_5;

void EnterStopMode(void) {
	/*Start an alarm for 10s*/
	setCursor(0,2);
	print("Entering STOP mode");
	is_stopped = 1;
	RTC_AlarmTypeDef sAlarm = {0};
    RTC_TimeTypeDef now;
    HAL_RTC_GetTime(&hrtc, &now, RTC_FORMAT_BIN);
    sAlarm.AlarmTime.Hours = now.Hours;
    sAlarm.AlarmTime.Minutes = now.Minutes;
    uint8_t secs = now.Seconds + 10;
    if (secs >= 60) {
  	    secs -= 60;
	    sAlarm.AlarmTime.Minutes = (now.Minutes + 1) % 60;
    } else {
	    sAlarm.AlarmTime.Minutes = now.Minutes;
    }
    sAlarm.AlarmTime.Seconds = secs;
    sAlarm.Alarm = RTC_ALARM_A;
    HAL_RTC_SetAlarm_IT(&hrtc, &sAlarm, RTC_FORMAT_BIN);

    /*Stop mode*/
//	HAL_SuspendTick();
//	HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
}

float Read_Temperature(void)
{
	float temperature = 0;
	int8_t rv;
	rv = r_single_Si7021(&temperature, Temperature);

	if (rv >= 0) {
		cdc_print("Temperature: %.1f C\r\n", temperature);
	}

	return temperature;
}

void SendStatsToTx(int temp) {
	uint8_t buf[10];
	snprintf((char*)buf, sizeof(buf), "%d", (temp << 8) | temp_setpt);
	if (LoRa_transmit(&myLoRa, buf, sizeof(buf) - 1, 2000)) {
		cdc_print("Sent stats to transmitter\n\r");
	}
	else {
		cdc_print("Transmission failed.");
	}
}

void LCD_UpdateTemp(int temp) {
    char num_str[20];
    setCursor(16, 0);

    if (temp < 10) {
        sprintf(num_str, " %d", temp);
    } else {
        sprintf(num_str, "%d", temp);
    }

    print(num_str);
    write(0xDF);  // Degree symbol
    print("C");
}

void LCD_UpdateSetpt(int setpt) {
    char num_str[20];
    setCursor(6, 0);

    if (setpt < 10) {
        sprintf(num_str, " %d", setpt);
    } else {
        sprintf(num_str, "%d", setpt);
    }

    print(num_str);
    write(0xDF);  // Degree symbol
    print("C");
}


void LCD_UpdateAirMode(size_t air_mode) {
    char heat_str[] = "Heat On";
    char ac_str[] = "Cool On";
    char off_str[] = "  Off  ";

    if (air_mode == HEAT) {
        setCursor((LCD_COLS / 2) - strlen(heat_str) / 2,3);
        print(heat_str);
    } else if (air_mode == AC) {
        setCursor((LCD_COLS / 2) - strlen(ac_str) / 2,3);
        print(ac_str);
    } else if (air_mode == OFF) {
        setCursor((LCD_COLS / 2) - strlen(off_str) / 2,3);
        print(off_str);
    }
}

void LCD_StrInit(void) {
    char setpt_str[] = "Setpt:";
    char temp_str[] = "T:";
    char off_str[] = "Off";

    // Setpt
    setCursor(0, 0);
    print(setpt_str);

    // Temp
    setCursor(14, 0);
    print(temp_str);

    // Air mode
    setCursor((LCD_COLS / 2) - strlen(off_str) / 2, 3);
    print(off_str);

    LCD_UpdateTemp((int) temperature);
    LCD_UpdateSetpt((int) temp_setpt);
    LCD_UpdateAirMode(air_mode);
}

void temp_change(size_t air_mode) {
	uint8_t buf[10];
	cdc_print("Temp_flag: %d\n\r", temp_flag);
	if (air_mode == AC) {
		if (temperature > temp_setpt) {
			// PullAcPinOn;
			if (temp_flag == 1) {
				pull_ac(GPIO_PIN_SET);
				cdc_print("Pulled AC pin HIGH\n\r");
			}
			temperature--;
			LCD_UpdateTemp((int) temperature);
			snprintf((char*)buf, sizeof(buf), "%d", (int) temperature);
			SendStatsToTx((int) temperature);
			cdc_print("Temperature: %.2f\r\n", temperature);
		}
		else {
			pull_ac(GPIO_PIN_RESET);
			cdc_print("Pulled AC pin LOW\n\r");
			temp_flag = 0;
			return;
		}
	}
	if (air_mode == HEAT) {
		if (temperature < temp_setpt) {
			// PullHeatPinOn;
			if (temp_flag == 1) {
				pull_heater(GPIO_PIN_SET);
				cdc_print("Pulled HEAT pin HIGH\n\r");
			}
			temperature++;
			LCD_UpdateTemp((int) temperature);
			snprintf((char*)buf, sizeof(buf), "%d", (int) temperature);
			SendStatsToTx((int) temperature);
			cdc_print("Temperature: %.2f\r\n", temperature);
			}
		else {
			pull_heater(GPIO_PIN_RESET);
			cdc_print("Pulled HEAT pin LOW\n\r");
			temp_flag = 0;
			return;
		}
	}
	cdc_print("Temperature adjusted.\n\r");
}

void pull_heater(int PIN_TYPE) {
	temp_flag = 2;
	HAL_GPIO_WritePin(HEAT_GPIO_PORT, HEAT_GPIO_PIN, PIN_TYPE);
}

void pull_ac(int PIN_TYPE) {
	temp_flag = 2;
	HAL_GPIO_WritePin(AC_GPIO_PORT, AC_GPIO_PIN, PIN_TYPE);
}

{
	if (GPIO_Pin == DIO0_Pin) {
		LoRa_receive(&myLoRa, RxBuffer, 16);
	}
}
